\def\optionalpagematter{
	\begin{itemize}
		\item Los ejercicios aquí planteados son para que practique lo visto en clase. Resuelva las ecuaciones diferenciales dadas (si se dan las condiciones iniciales o de frontera, encuentre la solución explicita). Grafique las soluciones con la ayuda de software. Señale la respuesta correcta. El puntaje obtenido en este cuestionario no representara su calificación, para que sea valido deberá estar acompañado con las operaciones necesarias y sus respectivas comprobaciones mediante software.
		\item Los ejercicios se dividieron en tres secciones:
		\begin{enumerate}
			\item[i)] Variables separables.
			\item[ii)] Ecuaciones diferenciales lineales (coeficientes constantes y factor integrante).
			\item[iii)] Ecuaciones diferenciales exactas.
			\item[iv)] Solución por sustitución (ED homogéneas, ecuación de Bernoulli y reducción a separación de variables).
			\item[v)] Aproximación numérica de soluciones de ED de primer orden (método de Runge-Kutta).
		\end{enumerate}
		\item Para comenzar cada sección, siga las siguientes instrucciones:
		\begin{enumerate}
			\item Primero haga clic en ``Begin quizz''.
			\item Marque las respuestas correctas y haga clic en ``End quizz'' (podrá ver el número de respuestas correctas).
			\item Haga clic en ``Correct'' y verá las respuestas correctas
		\end{enumerate}
	\end{itemize}
}

	\begin{quiz}{pronum1}{\textbf{i)} Variables separables.}

		{\BC{}\S{N}\Ff{\FfReadOnly}}

		\begin{questions}
			\begin{multicols}2
				
				\problem{$\frac{dy}{dx} = \sin(5x)$}%1
				\begin{answers}{1}
					\Ans0 $\frac{1}{5}\cos(5x)+c_1$
					\Ans1 $-\frac{1}{5}\cos(5x)+c_1$
					\Ans0 $-5\cos(5x)+c_1$
				\end{answers}
				
				
				\problem{$y' =\frac{y^2-1}{x^2-1}, \,\ y(2)=2$}%2
				\begin{answers}{1}
					\Ans1 $x$
					\Ans0 $-x$
					\Ans0 $x+c_1$
				\end{answers}
				
				
				\problem{$dx + e^{3x}dy = 0$}%3
				\begin{answers}{1}
					\Ans0 $\frac{e^{3x}}{3}+c_1$
					\Ans1 $\frac{e^{-3x}}{3}+c_1$
					\Ans0 $-\frac{e^{3x}}{3}+c_1$
				\end{answers}
				
				
				\problem{$dy - (y - 1)^2 dx = 0$}%4
				\begin{answers}{1}
					\Ans1 $\frac{x-1+c_1}{x+c_1}$
					\Ans0 $\frac{x+c_1}{x-1+c_1}$
					\Ans0 $\frac{x+c_1}{x+1+c_1}$
				\end{answers}
				
				
				\problem{$\frac{dx}{dt} = 4 (x^2 + 1), \,\ x(\pi/4) = 1$}%5
				\begin{answers}{1}
					\Ans0 $\tan\left(\frac{1}{3}(16x-4\pi)\right)$
					\Ans0 $\tan\left(\frac{1}{3}(16x+4\pi)\right)$
					\Ans1 $\tan\left(\frac{1}{4}(16x-3\pi)\right)$
				\end{answers}
				
				
				\problem{$x^2 y' = y - x y, \,\ y(-1) = -1$}%6
				\begin{answers}{1}
					\Ans0 $\frac{e^{1 -\frac{1}{x}}}{x}$
					\Ans1 $\frac{e^{-\frac{1}{x}-1}}{x}$
					\Ans0 $\frac{e^{\frac{1}{x}-1}}{x}$
				\end{answers}
				
				
				\problem{$e^x y y'=e^{-y-2x}+e^{-y}$}%7
				\begin{answers}{1}
					\Ans1 $e^y(y-1) = -\frac{1}{3}e^{-3x}-e^{-x} + c_1$
					\Ans0 $e^y(y-1) = \frac{1}{3}e^{-3x}-e^{-x} + c_1$
					\Ans0 $e^y(y-1) = \frac{1}{3}e^{-3x}+e^{-x} + c_1$
				\end{answers}
				
				
				\problem{$\left(x^4+1\right)dy+x\left(4y^2+1\right)dx=0, \,\ y(1)=0$}%8
				\begin{answers}{1}
					\Ans0 $\frac{1}{2}\tan\left(\frac{\pi}{4}-\tan^{-1}\left(x\right)\right)$
					\Ans0 $\frac{1}{2}\tan\left(\frac{\pi}{4}+\tan^{-1}\left(x^2\right)\right)$
					\Ans1 $\frac{1}{2}\tan\left(\frac{\pi}{4}-\tan^{-1}\left(x^2\right)\right)$
				\end{answers}
				
				
				\problem{$\sin(3x)dx + 2y\cos^3(3x)dy = 0$}%9
				\begin{answers}{1}
					\Ans0 $\sqrt{-\frac{1}{6}\sec^2(3x)+2c_1}$
					\Ans1 $-\sqrt{-\frac{1}{6}\sec^2(3x)+2c_1}$
					\Ans0 $-\sqrt{\frac{1}{6}\sec^2(3x)+2c_1}$
				\end{answers}
				
				
				\problem{$y'(x)=\frac{2x+1}{2y(x)}, \,\ y(-2)=-1$}%10
				\begin{answers}{1}
					\Ans0 $\sqrt{x^2+x-1}$
					\Ans0 $\sqrt{x^2+x+1}$
					\Ans1 $-\sqrt{x^2+x-1}$
				\end{answers}
				
			\end{multicols}
		\end{questions}

	\begin{quiz}{pronum2}{\textbf{ii)} Ecuaciones diferenciales lineales (coeficientes constantes y factor integrante)}
	
	\def\AnswerFieldDefaults
	{\BC{}\S{N}\Ff{\FfReadOnly}}
	\def\problem#1\par{\bigskip\begin{minipage}{\linewidth}\item #1\end{minipage}}
	
	\begin{questions}
		\begin{multicols}2
			
			\problem{$3 \frac{dy}{dx} + 12 y = 4$}%1
			\begin{answers}{1}
				\Ans0 $\frac{1}{3}-c_1e^{-4x}$
				\Ans0 $\frac{1}{3}+c_1e^{4x}$
				\Ans1 $\frac{1}{3}+c_1e^{-4x}$
			\end{answers}
			
			
			\problem{$x y' + y = e^x, \,\ y(1) = 2$}%2
			\begin{answers}{1}
				\Ans1 $\frac{e^x+2-e}{x}$
				\Ans0 $\frac{e^x+2+e}{x}$
				\Ans0 $\frac{-e^x+2-e}{x}$
			\end{answers}
			
			
			\problem{$\frac{dy}{dx} = 5 y$}%3
			\begin{answers}{1}
				\Ans0 $-c_1e^{5x}$
				\Ans0 $c_1e^{-5x}$
				\Ans1 $c_1e^{5x}$
			\end{answers}
			
			
			\problem{$x y'- y = x^2 \sin(x)$}%4
			\begin{answers}{1}
				\Ans0 $x(\cos(x)+c_1)$
				\Ans1 $x(-\cos(x)+c_1)$
				\Ans0 $-x(\cos(x)+c_1)$
			\end{answers}
			
			
			\problem{$y \frac{dx}{dy} - x = 2 y^2, \,\ x(1) = 5$}%5
			\begin{answers}{1}
				\Ans0 $2y^2-3y$
				\Ans0 $-2y^2+3y$
				\Ans1 $2y^2+3y$
			\end{answers}
			
			
			\problem{$\frac{dr}{d\theta} -r \sec(\theta) = \cos(\theta)$}%6
			\begin{answers}{1}
				\Ans0 $(\theta+\cos(\theta)+c_1)e^{-\theta}$
				\Ans1 $(\theta+\cos(\theta)+c_1)e^\theta$
				\Ans0 $(\theta-\cos(\theta)+c_1)e^\theta$
			\end{answers}
			
			
			\problem{$(x+1)y'(x)+y(x)=\log(x), \,\ y(1) = 10$}%7
			\begin{answers}{1}
				\Ans1 $\frac{-x+x\log(x)+21}{x+1}$
				\Ans0 $\frac{x+x\log(x)+21}{x-1}$
				\Ans0 $\frac{x+x\log(x)+21}{x+1}$
			\end{answers}
			
			
			\problem{$\cos(x)\frac{dy}{dx} + y \sin(x) = 1$}%8
			\begin{answers}{1}
				\Ans0 $-\sin(x)+c_1\cos(x)$
				\Ans0 $\sin(x)-c_1\cos(x)$
				\Ans1 $\sin(x)+c_1\cos(x)$
			\end{answers}
			
			
			\problem{$cos(x)\frac{dy}{dx} + y \sin(x) = x$}%9
			\begin{answers}{1}
				\Ans0 $x\sin(x)-\cos(x)(\log(\cos(x))+c_1)$
				\Ans1 $x\sin(x)+\cos(x)(\log(\cos(x))+c_1)$
				\Ans0 $-x\sin(x)+\cos(x)(\log(\cos(x))+c_1)$
			\end{answers}
			
		
			\problem{$\frac{dy}{dx} + 2xy = f(x),  \,\ y(0)=2,\\
				\\
				f(x) = \left\{
				x, \quad\ 0 \leq x < 1 \atop
				0, \qquad\ t \geq 1
				\right.$}%10
			\begin{answers}{1}
				\Ans0 $\frac{1}{2}e^{-x^2}\left(\left(e^{-x^2}-e\right)\theta(1-x)+e+3\right)$
				\Ans0 $\frac{1}{2}e^{x^2}\left(\left(e^{x^2}-e\right)\theta(1-x)+e+3\right)$
				\Ans1 $\frac{1}{2}e^{-x^2}\left(\left(e^{x^2}-e\right)\theta(1-x)+e+3\right)$
			\end{answers}
			
		\end{multicols}
	\end{questions}
\end{quiz}
